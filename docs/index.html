@vbarbarosh/dd
An abstraction for working with the mouse.

{{ p.x }},{{ p.y }} 
<div v-on:mousedown="mousedown($event, p)" v-bind:style="{top: px(p.y), left: px(p.x)}" class="box">
    {{ p.x }},{{ p.y }}
</div>

function mousedown(event, p)
{
    const x0 = p.x;
    const y0 = p.y;
    dd({
        event,
        move: function ({dx, dy}) {
            p.x = x0 + dx;
            p.y = y0 + dy;
        },
    });
}
